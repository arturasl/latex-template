%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Lithuanian

% automatically rename document elements (contents <-> turinys, etc.)
\usepackage[lithuanian]{babel}

% fixing captions
\captionsetup[table]{labelsep=space}      % do not add colon after table names (Table.: <- colon is bad)
\captionsetup[figure]{labelsep=space}     % do not add colon after figure names (Pav.: <- colon is bad)
\captionsetup[lstlisting]{labelsep=space} % also no column after lstlisting names
\captionsetup[listing]{labelsep=space}    % also no column after minted code blocks names

% use same labels for lstlistings as for other figures (Pav. insteand of Listing:)
\def\lstlistingname{\figurename}
\def\lstlistlistingname{\figurename}
\def\lstlistingautorefname{\figurenameautorefname}

% label below mited code blocks, should be same as figure names
\def\listingscaption{\figurename}

% commands for writing shorthands
\newcommand{\eng}[1]{(angl. \emph{#1})}
\newcommand{\shortform}[1]{(toliau #1)}
\newcommand{\engshortform}[2]{(angl. \emph{#1}, toliau #2)}

% distance between figures and captions
\setlength{\abovecaptionskip}{4pt}
\setlength{\belowcaptionskip}{4pt}
\setlength{\intextsep}{8pt plus 1pt minus 1pt}
% \setlength{\textfloatsep}{5pt plus 1pt minus 1pt}

% glossaries
\renewcommand*{\glossaryname}{Sąvokų apibrėžimai}
\renewcommand*{\acronymname}{Santrumpos}
